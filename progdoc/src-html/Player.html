001import java.util.Stack;
002/**
003 * This class is part of the "World of Zuul" application. 
004 * "World of Zuul" is a very simple, text based adventure game.  
005 * This class is used to initialize players and manage everything related to players.
006 */
007public class Player
008{
009   private Room aCurrentRoom;
010   private String aName;
011   private double aPoids;
012   private double aPoidsMax;
013   private Stack<Room> aPrecedRooms;
014   private ItemList aItemsPlayer;
015   
016   /**
017    * Constructeur qui initialise les attributs de la classe Player
018    * @param Room de départ
019    * @param String nom du joueur
020    */
021   public Player(final Room pRoom, final String pName)
022   {
023       this.aCurrentRoom = pRoom;
024       this.aName = pName;
025       this.aPoids = 0;
026       this.aPoidsMax = 7; 
027       this.aPrecedRooms = new Stack<Room>();
028       this.aItemsPlayer = new ItemList();
029   }
030   
031   /**
032    * Accesseur qui permet d"accéder à aCurrentRoom
033    * @return la room courante
034    */
035   public Room getCurrentRoom()
036   {
037       return this.aCurrentRoom;
038   }
039   
040   /**
041    * Modificateur de aCurrentRoom
042    * @param Room valeur que prend aCurrentRoom 
043    */
044   public void setCurrentRoom(final Room pRoom)
045   {
046       this.aCurrentRoom = pRoom;
047   }
048   
049   /**
050    * Accesseur qui permet d'accéder au nom
051    * @return le nom du player
052    */
053   
054   public String getName()
055   {
056       return this.aName;
057   }
058   
059   /**
060    * Accesseur qui permet d'accéder au poids actuelle du joueur 
061    * @return le poids que porte le player
062    */
063   public double getPoids()
064   {
065       return this.aPoids;
066   }
067   
068   /**
069    * Accesseur qui permet d'accéder au poids maximum que le joueur peut porter
070    * @return le poids maximum que peut porter le player
071    */
072   public double getPoidsMax()
073   {
074       return this.aPoidsMax;
075   }
076   
077   /**
078    * Modificateur de aPoidsMax
079    * @param double nouvelle valeur que prend aPoidsMax
080    */
081   public void setPoidsMax(final double pPoidsMax)
082   {
083       this.aPoidsMax = pPoidsMax;
084   }
085   
086   /**
087    * Fonction qui renvoie et supprime le dernier élément de la pile aPrecedRooms
088    * @return le dernier élément de la pile
089    */
090   public Room getPrecedRoom()
091   {
092       return this.aPrecedRooms.pop();
093   }
094   
095   /**
096    * Procédure qui ajoute une room dans la pile aPrecedRooms
097    * @param Room que l'on veut ajouter
098    */
099   public void ajoutePrecedRooms(final Room pPrecedRoom)
100   {
101       this.aPrecedRooms.push(pPrecedRoom);
102   }
103   
104   /**
105    * Fonction qui vérifie si la pile aPrecedRooms est vide ou pas
106    * @return true si la pile est vide false sinon
107    */
108   public boolean precedRoomsEmpty()
109   {
110       return this.aPrecedRooms.empty();
111   }
112   
113   /**
114    * Fonction qui permet de récuperer l'item portant le nom passé en paramètre
115    * @param String de l'item que l'on souhaite récupérer
116    * @return l'item correspondant au nom passer en paramètre
117    */
118   public Item getItemPlayer(final String pNom)
119   {
120       return this.aItemsPlayer.getItemList(pNom);
121   }
122   
123   /**
124    * Procédure qui vérifie si le joueur peut prendre un item passé en paramètre
125    * @return vrai si oui, false sinon
126    */
127   public boolean canTake(final Item pItem)
128   {
129       return this.aPoids + pItem.getPoids() <= this.aPoidsMax;
130   }
131   
132   /**
133    * Procédure qui permet de ramasser l'item passé en paramètre
134    * @param Item que l'on souhaite ramasser
135    */
136   public void takeItem(final Item pItem)
137   {
138       this.aItemsPlayer.putItemList(pItem);
139       this.aPoids = this.aPoids + pItem.getPoids();
140   }
141   
142   /**
143    * Procédure qui permet de jeter l'item passé en paramètre
144    * @param Item que l'on souhaite jeter
145    */
146   public void dropItem(final Item pItem)
147   {
148       this.aItemsPlayer.removeItemList(pItem);
149       this.aPoids = this.aPoids - pItem.getPoids();
150   }
151    
152   /**
153    * Fonction qui retourne tous les items que posséde le joueur
154    * @return une String des items que posséde le joueur
155    */
156   public String getItemPlayerString()
157   {
158       return this.aItemsPlayer.getItemListString();
159   }
160   
161   /**
162    * Fonction qui dit si la ItemList du joueur est vide
163    * @return true si c'est vide false sinon
164    */
165   public boolean itemIsEmpty()
166   {
167       return this.aItemsPlayer.itemListIsEmpty();
168   }
169   
170   /**
171    * Fonction qui renvoie l'élément de tête de la pile sans la supprimer
172    * @retunr l'élément de tête de la pile
173    */
174   public Room peekPrecedRooms()
175   {
176       return this.aPrecedRooms.peek();
177   }
178   
179   /**
180    * Procédure qui supprime tout les éléments de la pile
181    */
182   public void clearPrecedRooms()
183   {
184       this.aPrecedRooms.clear();
185   }
186}
