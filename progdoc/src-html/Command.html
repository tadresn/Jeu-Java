001  /**
002 * This class is part of the "World of Zuul" application. 
003 * "World of Zuul" is a very simple, text based adventure game.  
004 * This class is used to initialize commands and manage everything related to commands.
005 */
006 public class Command
007{
008    private String aCommandWord;
009    private String aSecondWord;
010    
011    /**
012     * Constructeur naturel
013     * @param pCommandWord String pour initialiser aCommandWord
014     * @param pSecondWord String pour initialiser aSecondWord
015     */
016    public Command(final String pCommandWord, final String pSecondWord)
017    {
018       this.aCommandWord = pCommandWord;
019       this.aSecondWord = pSecondWord;
020    }
021    
022    /**
023     * Accesseur pour aCommandWord
024     * @return le premier mot
025     */
026    public String getCommandWord()
027    {
028       return this.aCommandWord;
029    }
030    
031    /**
032     * Accesseur pour aSecondWord
033     * @return le second mot
034     */
035    public String getSecondWord()
036    {
037        return this.aSecondWord;
038    }
039    
040    /**
041     * Permet de vérifier qu'un second mot a bien été tapé
042     * @return true si il y'a un second mot sinon elle renvoie false
043     */
044    public boolean hasSecondWord()
045    {
046        return this.aSecondWord != null;
047    }
048    
049    /**
050     * Permet de vérifier si le premier mot est null
051     * @return true si le premier mot est null sinon elle renvoie false
052     */
053    public boolean isUnknown()
054    {
055        return this.aCommandWord == null;
056    }
057} // Command
