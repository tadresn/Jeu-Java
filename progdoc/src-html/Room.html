001import java.util.HashMap;
002import java.util.Set;
003
004/**
005 * This class is part of the "World of Zuul" application. 
006 * "World of Zuul" is a very simple, text based adventure game.  
007 * This class allows you to initialize rooms and manage everything related to rooms.
008 */
009
010 public class Room
011{
012    private String aDescription;
013    private HashMap<String, Room> aExits;
014    private String aImageName;
015    private ItemList aItems;
016    
017    /**
018     * Constructeur naturel qui crée une pièce décrite par la chaîne 'pDescription' et avec une image donnée
019     * @param pDescription String pour initialiser aDescription
020     * @param String du lien vers l'image
021     */
022    public Room(final String pDescription, final String pImage)
023    {
024        this.aDescription = pDescription;
025        this.aExits = new HashMap<String, Room>();
026        this.aImageName = pImage;
027        this.aItems = new ItemList();
028    }
029    
030    /**
031     * Accesseur pour aDescription
032     * @return la description de la pièce
033     */
034    public String getDescription()
035    {
036        return this.aDescription;
037    }
038    
039    /**
040     * Procédure qui permet d'ajouter les sorties de la room.
041     * @param pDirection String la direction de la sortie
042     * @param pRoom Room la salle dans la direction donnée
043     */
044    public void setExit(final String pDirection, final Room pRoom)
045    {
046        aExits.put(pDirection, pRoom);
047    }
048    
049    /**
050     * Procédure qui permet d'ajouter les items dans les rooms
051     * @param pItem Item à ajouter 
052     */
053    public void addItem(final Item pItem)
054    {
055        this.aItems.putItemList(pItem);
056    }
057    
058    /**
059     * Procédure qui permet de supprimer l'item passé en paramètre
060     * @param Item que l'on souhaite supprimer
061     */
062    public void removeItem(final Item pItem)
063    {
064        this.aItems.removeItemList(pItem);
065    }
066    
067    /**
068     *Fonction qui retourne l'item portant le nom passé en paramètre
069     *@param String de l'item que l'on souhaite récupérer
070     */
071    public Item getItemRoom(final String pNom)
072    {
073        return this.aItems.getItemList(pNom);
074    }
075    
076    /**
077     * Accesseur permettant d'accéder à la sortie
078     * @return la pièce atteinte si nous nous déplaçons dans la 
079     * direction 'pDirection'
080     */
081    public Room getExit(final String pDirection)
082    {
083       return this.aExits.get(pDirection);
084    }
085        
086    /**
087     * Fonction qui retourne les sortie disponible de la CurrentRoom
088     * @return les sorties disponibles
089     */
090    public String getExitString()
091    {
092    String vS = "Exits :";
093    Set<String> vKeys = this.aExits.keySet();
094    for (String vExit : vKeys) {
095        vS += " " + vExit;
096    }
097    return vS;
098    }
099    
100    /**
101     * Fontion qui retourne les items disponible dans la CurrentRoom
102     * @return les items disponibles
103     */
104    public String getItemString()
105    {
106        return this.aItems.getItemListString();
107    }
108
109    /**
110     * Renvoie une description détaillée de cette pièce
111     * @return Une description de la pièce avec les sorties et les items
112     */
113    public String getLongDescription()
114    {
115        if (this.aItems.itemListIsEmpty()) {
116        return "You are " + this.aDescription + "\n" + this.getExitString()
117        + "\nThere is no item here.";
118       }
119       else {
120        return "You are " + this.aDescription + "\n" + this.getExitString()
121        + "\n" + getItemString();
122       }
123    }
124           
125    /**
126     * Return une string décrivant le nom de l'image de la pièce
127     */
128    public String getImageName()
129    {
130        return this.aImageName;
131    }
132    
133    /**
134     * Fonction qui renvoie vrai ou faux selon que la Room passée en paramètre est une des sorties de la pièce ou pas.
135     * @return true si la Room passée en paramètre est une sortie de la pièce et false sinon
136     */
137    public boolean isExit(final Room pRoom)
138    {
139        return this.aExits.containsValue(pRoom);
140    }
141            
142}// Room
