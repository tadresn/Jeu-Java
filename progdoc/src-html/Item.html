002/**
003 * This class is part of the "World of Zuul" application. 
004 * "World of Zuul" is a very simple, text based adventure game. 
005 * This class is used to initialize items and manage everything related to items.
006 */
007public class Item
008{
009  private String aNom;
010  private double aPoids;
011  
012  /**
013   * Constructeur de la classe Item
014   * @param String le nom, c'est-à-dire la description de l'item
015   * @param double le poids de l'item
016   */
017  public Item(final String pNom, final double pPoids)
018  {
019      this.aNom = pNom;
020      this.aPoids = pPoids;
021  }
022  
023  /**
024   * Accesseur qui permet d'accéder au nom de l'item.
025   * @return le nom de l'item
026   */
027  public String getNom()
028  {
029      return this.aNom;
030  }
031  
032  /**
033   * Accesseur qui permet d'accéder au poids de l'item.
034   * @return le poids de l'item 
035   */
036  public double getPoids()
037  {
038      return this.aPoids;
039  }
040  
041}
