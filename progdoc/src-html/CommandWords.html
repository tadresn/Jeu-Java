001 /**
002 * This class is part of the "World of Zuul" application. 
003 * "World of Zuul" is a very simple, text based adventure game.  
004 * 
005 * This class holds an enumeration table of all command words known to the game.
006 * It is used to recognise commands as they are typed in.
007 *
008 * @author  Michael Kolling and David J. Barnes + D.Bureau
009 * @version 2008.03.30 + 2019.09.25
010 */
011public class CommandWords
012{
013    // a constant array that will hold all valid command words
014    private static final String[] VALIDCOMMANDS = {"go","quit","help","look","take","back","test", "drop", "inventaire", "eat", "charge", "fire"};
015
016    /**
017     * Constructor - initialise the command words.
018     */
019    public CommandWords()
020    {
021        // rien
022    }
023    
024    /**
025     * Check whether a given String is a valid command word. 
026     * @return true if a given string is a valid command,
027     * false if it isn't.
028     */
029    public boolean isCommand( final String pString )
030    {
031        for ( int i=0; i<VALIDCOMMANDS.length; i++ ) {
032            if ( VALIDCOMMANDS[i].equals( pString ) )
033                return true;
034        } // for
035        // if we get here, the string was not found in the commands
036        return false;
037    } // isCommand()
038    
039    /**
040     * Retourne toutes les commandes valides
041     * @return les commandes valides
042     */
043    public String getCommandList()
044    {
045        String vS = " ";
046        for (String pCommand : VALIDCOMMANDS){
047            vS = vS + pCommand + " " ;
048        }
049        return vS;
050    }
051        
052} // CommandWords
