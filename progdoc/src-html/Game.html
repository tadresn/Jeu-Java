001/**
002 * This class is part of the "World of Zuul" application. 
003 * "World of Zuul" is a very simple, text based adventure game. 
004 * It allows you to create the game and intialize its internal card as well as its interface.
005 */
006 public class Game
007{
008    private UserInterface aGui;
009    private GameEngine aEngine;
010    
011    /**
012     * Crée le jeu et initialise sa carte interne. 
013     * Crée l'interface et crée un lien vers celle-ci.
014     */
015    public Game()
016    {
017        this.aEngine = new GameEngine();
018        this.aGui = new UserInterface(this.aEngine);
019        this.aEngine.setGUI(this.aGui);
020    }
021 
022}// Game
