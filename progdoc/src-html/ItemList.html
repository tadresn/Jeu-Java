001import java.util.HashMap;
002import java.util.Set;
003/**
004 * This class is part of the "World of Zuul" application. 
005 * "World of Zuul" is a very simple, text based adventure game.
006 * This class is used to manage lists of items.
007 */
008public class ItemList
009{
010    private HashMap<String, Item> aItemList;
011    
012    /**
013     * Constructeur de la classe ItemList
014     */
015    public ItemList()
016    {
017        this.aItemList = new HashMap<String, Item>();
018    }
019    
020    /**
021     * Fonction qui retourne l'item portant le nom passé en paramètre
022     * @param String de l'item que l'on souhaite récupérer
023     * @return l'item correspondant au nom passer en paramètre
024     */
025    public Item getItemList(final String pNom)
026    {
027        return this.aItemList.get(pNom);
028    }
029    
030    /**
031     * Procédure qui permet d'ajouter l'item passé en paramètre
032     * @param Item que l'on souhaite ajouter dans la ItemList
033     */
034    public void putItemList(final Item pItem)
035    {
036        this.aItemList.put(pItem.getNom(), pItem);
037    }
038    
039    /**
040    * Procédure qui permet de retirer l'item passé en paramètre
041    * @param Item que l'on souhaite retirer de la ItemList
042    */
043    public void removeItemList(final Item pItem)
044    {
045        this.aItemList.remove(pItem.getNom());
046    }
047    
048    
049     /**
050     * Fontion qui retourne les items sous forme de String
051     * @return les items présent dans ItemList
052     */
053    public String getItemListString()
054    {
055        String vS = "Items :";
056        Set <String> vKeys = this.aItemList.keySet();
057        for (String vItem : vKeys) {
058            vS += " " + vItem;
059        }
060
061        return vS;
062    }
063    
064    /**
065     * Fonction qui vérifie si l'ItemList est vide
066     * @return vrai si l'ItemList est vide, faux sinon
067     */
068    public boolean itemListIsEmpty()
069    {
070        return this.aItemList.isEmpty();
071    }
072 
073}
